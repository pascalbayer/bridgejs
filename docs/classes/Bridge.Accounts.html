  • password
  • [callback]
)

Defined in server/lib/lib.bridge.js:851

USAGE: Client
Login a user by providing username and password, if the login fails a single Error object is passed to the callback function as parameter

Parameters:

  • user String
  • password String
  • [callback] Function optional

Example:

Login a user

 Bridge.Accounts.loginWithPassword('username', 'password', function (Error) {
     if (Error) {
         console.log(Error.message);
     }
 });

validateNewUser

(
  • callback
)

Defined in server/lib/lib.bridge.js:806

USAGE: Server
Validates the creation of a new user account, returns true by default. Specify a callback to perform a specific validation returning true or false.

Parameters:

  • callback Function

Example:

Register a callback to customize the validation of a new user account creation

 Bridge.Accounts.validateNewUser(function (user) {
     if (usernameIsAvailable(user.username)) {
         return true;
     }
     else {
         return false;
     }
 });
